async function loadThemes(forceReseed = false) {
  if (forceReseed) {
    if (!confirm('Reset Themes will wipe all article-theme assignments and re-analyse your entire library from scratch.\n\nThis takes 60-90 seconds and costs ~$0.07.\n\nContinue?')) return;
    await seedThemes();
    return;
  }
  const kt = document.getElementById('kt-body');
  try {
    const resp = await fetch(SERVER + '/api/kt/themes', {cache: 'no-store'});
    const data = await resp.json();
    if (data.seeded && data.themes && data.themes.length > 0) {
      ktThemes = data.themes;
      ktSelectedIdx = null;
      renderThemeGrid(ktThemes);
      checkEvolution();
      return;
    }
    // Not seeded yet — show seed prompt
    showSeedPrompt(kt);
  } catch(e) {
    if (kt) kt.innerHTML = '<div class="kt-generate-state"><div class="kt-generate-title">Could not reach server</div><div class="kt-generate-sub" style="color:var(--red)">' + e.message + '</div></div>';
  }
}

function showSeedPrompt(container) {
  if (!container) return;
  const n = (articles || []).length;
  container.innerHTML = `
    <div class="kt-generate-state">
      <div style="font-size:48px;margin-bottom:8px;">🧠</div>
      <div class="kt-generate-title">Key Themes not yet generated</div>
      <div class="kt-generate-sub">Meridian will analyse your ${n} articles and identify 10 dominant intelligence themes.<br>Runs once — themes update incrementally after each sync.</div>
      <button class="kt-generate-btn" onclick="seedThemes()">✦ Generate Themes</button>
    </div>`;
}

async function seedThemes() {
  const kt = document.getElementById('kt-body');
  if (kt) kt.innerHTML = `
    <div class="kt-generate-state">
      <div style="font-size:36px;margin-bottom:8px;">⏳</div>
      <div class="kt-generate-title" id="seed-progress-msg">Starting seed…</div>
      <div class="kt-generate-sub">Analysing ${(articles || []).length} articles with Claude Sonnet…</div>
      <div style="width:300px;height:4px;background:#e8e0d4;border-radius:2px;overflow:hidden;margin-top:12px;">
        <div id="seed-progress-bar" style="height:100%;width:8%;background:rgb(196,120,58);border-radius:2px;transition:width 2s;"></div>
      </div>
    </div>`;

  let jobId;
  try {
    const r = await fetch(SERVER + '/api/kt/seed', {method:'POST', headers:{'Content-Type':'application/json'}, body:'{}'});
    const j = await r.json();
    if (!j.ok) throw new Error(j.error || 'Seed failed to start');
    jobId = j.job_id;
  } catch(e) {
    if (kt) kt.innerHTML = '<div class="kt-generate-state"><div class="kt-generate-title">Seed failed</div><div class="kt-generate-sub" style="color:var(--red)">' + e.message + '</div><button class="kt-generate-btn" onclick="seedThemes()">↻ Try again</button></div>';
    return;
  }

  let width = 8;
  const poll = setInterval(async () => {
    try {
      const r = await fetch(SERVER + '/api/kt/seed/status/' + jobId, {cache:'no-store'});
      const j = await r.json();
      const msg = document.getElementById('seed-progress-msg');
      const bar = document.getElementById('seed-progress-bar');
      if (msg) msg.textContent = j.progress || 'Working…';
      width = Math.min(width + 7, 92);
      if (bar) bar.style.width = width + '%';
      if (j.status === 'done') {
        clearInterval(poll);
        if (bar) bar.style.width = '100%';
        if (msg) msg.textContent = 'Complete! Loading themes…';
        setTimeout(() => loadThemes(false), 800);
      } else if (j.status === 'error') {
        clearInterval(poll);
        const kt2 = document.getElementById('kt-body');
        if (kt2) kt2.innerHTML = '<div class="kt-generate-state"><div class="kt-generate-title">Seed error</div><div class="kt-generate-sub" style="color:var(--red)">' + (j.error || 'Unknown error') + '</div><button class="kt-generate-btn" onclick="seedThemes()">↻ Try again</button></div>';
      }
    } catch(e) { /* retry */ }
  }, 3000);
}
